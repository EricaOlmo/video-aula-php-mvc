	id int primary key,
	nome varchar(45),
	login varchar(45),
	senha varchar(32),
	status char(1)	
);

create table if not exists empresa
(
id integer not null primary key auto_increment,
nome varchar (45),
situacao varchar (25)
);

create table if not exists categoria
(
id integer not null primary key auto_increment,
descricao varchar (45),
situacao varchar(25)
);

create table if not exists produto
(
id integer not null primary key auto_increment,
nome varchar (45),
valor float (10),
situacao varchar (25)
);
